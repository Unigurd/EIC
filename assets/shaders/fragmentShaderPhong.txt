#version 430 core

in vec4 norm;
uniform vec3 color;
uniform vec3 lightDir;
uniform vec3 lightColor;
uniform float ambientStrength;
out vec4 FragColor;

void main()
{
    vec3 nld = normalize(vec3(1.0,-0.8,1.0));
    vec3 tmpNorm = normalize(vec3(1.0,0.1,-0.0));

    vec3 ambient = ambientStrength * lightColor;
    float diff = max(dot(normalize(norm.xyz), -normalize(lightDir)), 0.0);
    vec3 diffuse = diff * lightColor;
    vec3 result = (ambient + diffuse) * color;
    FragColor = vec4(result, 1.0);
}
